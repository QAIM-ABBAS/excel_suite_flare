   117‚Üí   117‚Üídef tool_merge(args):
   118‚Üí   118‚Üí    files = args.get("files", [])
   119‚Üí   119‚Üí    output_format = args.get("outputFormat", "xlsx")
   120‚Üí   120‚Üí    output_filename = args.get("outputFilename", "merged")
   121‚Üí   121‚Üí
   122‚Üí   122‚Üí    if len(files) < 2:
   123‚Üí   123‚Üí        return {"error": "At least 2 files are required"}
   124‚Üí   124‚Üí
   125‚Üí   125‚Üí    all_dfs = []
   126‚Üí   126‚Üí    all_headers = []
   127‚Üí   127‚Üí    for f in files:
   128‚Üí   128‚Üí        df = read_file_to_df(f)
   129‚Üí   129‚Üí        if not df.empty:
   130‚Üí   130‚Üí            all_headers.append(list(df.columns))
   131‚Üí   131‚Üí            all_dfs.append(df)
   132‚Üí   132‚Üí
   133‚Üí   133‚Üí    merged = pd.concat(all_dfs, ignore_index=True)
   134‚Üí   134‚Üí    base_headers = all_headers[0] if all_headers else []
   135‚Üí   135‚Üí    has_mismatch = any(len(h) != len(base_headers) or h != base_headers for h in all_headers)
   136‚Üí   136‚Üí
   137‚Üí   137‚Üí    safe_name = sanitize_filename(output_filename) or "merged"
   138‚Üí   138‚Üí    info = save_df_to_file(merged, safe_name, "", output_format)
   139‚Üí   139‚Üí    record_file(info["filename"], f"{safe_name}.xlsx", info["mime"], info["size"], "merge", info["filepath"])
   140‚Üí   140‚Üí
   141‚Üí   141‚Üí    return {
   142‚Üí   142‚Üí        "success": True,
   143‚Üí   143‚Üí        "downloadUrl": f"/api/tools/download?file={info['filename']}",
   144‚Üí   144‚Üí        "filename": info["filename"],
   145‚Üí   145‚Üí        "totalRows": len(merged),
   146‚Üí   146‚Üí        "headers": base_headers,
   147‚Üí   147‚Üí        "hasMismatch": has_mismatch,
   148‚Üí   148‚Üí        "mismatchWarning": "Some files have different headers. Data was merged using all available columns." if has_mismatch else None,
   149‚Üí   149‚Üí    }

   244‚Üí   244‚Üídef tool_stats(args):
   245‚Üí   245‚Üí    filepath = args.get("filepath", "")
   246‚Üí   246‚Üí    generate_report = args.get("generateReport", "false")
   247‚Üí   247‚Üí
   248‚Üí   248‚Üí    if not filepath:
   249‚Üí   249‚Üí        return {"error": "File is required"}
   250‚Üí   250‚Üí
   251‚Üí   251‚Üí    df = read_file_to_df(filepath)
   252‚Üí   252‚Üí    if df.empty:
   253‚Üí   253‚Üí        return {"error": "File is empty"}
   254‚Üí   254‚Üí
   255‚Üí   255‚Üí    stats = []
   256‚Üí   256‚Üí    for col in df.columns:
   257‚Üí   257‚Üí        values = df[col].tolist()
   258‚Üí   258‚Üí        non_empty = [v for v in values if v != "" and v is not None]
   259‚Üí   259‚Üí        numeric_vals = []
   260‚Üí   260‚Üí        for v in non_empty:
   261‚Üí   261‚Üí            try:
   262‚Üí   262‚Üí                numeric_vals.append(float(v))
   263‚Üí   263‚Üí            except (ValueError, TypeError):
   264‚Üí   264‚Üí                pass
   265‚Üí   265‚Üí
   266‚Üí   266‚Üí        distinct = len(set(str(v) for v in non_empty))
   267‚Üí   267‚Üí        missing = len(values) - len(non_empty)
   268‚Üí   268‚Üí
   269‚Üí   269‚Üí        if len(non_empty) == 0:
   270‚Üí   270‚Üí            col_type = "empty"
   271‚Üí   271‚Üí        elif len(numeric_vals) == len(non_empty):
   272‚Üí   272‚Üí            col_type = "numeric"
   273‚Üí   273‚Üí        elif len(numeric_vals) == 0:
   274‚Üí   274‚Üí            col_type = "text"
   275‚Üí   275‚Üí        else:
   276‚Üí   276‚Üí            col_type = "mixed"
   277‚Üí   277‚Üí
   278‚Üí   278‚Üí        stat = {"column": col, "type": col_type, "count": len(non_empty), "distinct": distinct, "missing": missing}
   279‚Üí   279‚Üí
   280‚Üí   280‚Üí        if col_type in ("numeric", "mixed") and numeric_vals:
   281‚Üí   281‚Üí            stat["sum"] = round(sum(numeric_vals), 4)
   282‚Üí   282‚Üí            stat["avg"] = round(statistics.mean(numeric_vals), 4)
   283‚Üí   283‚Üí            stat["min"] = min(numeric_vals)
   284‚Üí   284‚Üí            stat["max"] = max(numeric_vals)
   285‚Üí   285‚Üí            sorted_nums = sorted(numeric_vals)
   286‚Üí   286‚Üí            n = len(sorted_nums)
   287‚Üí   287‚Üí            if n % 2 == 0:
   288‚Üí   288‚Üí                stat["median"] = round((sorted_nums[n // 2 - 1] + sorted_nums[n // 2]) / 2, 4)
   289‚Üí   289‚Üí            else:
   290‚Üí   290‚Üí                stat["median"] = round(sorted_nums[n // 2], 4)
   291‚Üí   291‚Üí            if len(numeric_vals) > 1:
   292‚Üí   292‚Üí                stat["stdDev"] = round(statistics.stdev(numeric_vals), 4)
   293‚Üí   293‚Üí            else:
   294‚Üí   294‚Üí                stat["stdDev"] = 0.0
   295‚Üí   295‚Üí
   296‚Üí   296‚Üí        if col_type in ("text", "mixed") and non_empty:
   297‚Üí   297‚Üí            lengths = [len(str(v)) for v in non_empty]
   298‚Üí   298‚Üí            stat["minLength"] = min(lengths)
   299‚Üí   299‚Üí            stat["maxLength"] = max(lengths)
   300‚Üí   300‚Üí
   301‚Üí   301‚Üí        freq = Counter(str(v) for v in non_empty)
   302‚Üí   302‚Üí        stat["topValues"] = [{"value": v, "count": c} for v, c in freq.most_common(5)]
   303‚Üí   303‚Üí        stats.append(stat)

   393‚Üí   393‚Üí

   397‚Üí   397‚Üí
   398‚Üí   398‚Üí    for c in conditions:
   399‚Üí   399‚Üí        if c.get("column") not in df.columns:
   400‚Üí   400‚Üí            return {"error": f"Column '{c.get('column')}' not found"}
   401‚Üí   401‚Üí
   402‚Üí   402‚Üí    def apply_condition(row, cond):
   403‚Üí   403‚Üí        col = cond["column"]
   404‚Üí   404‚Üí        op = cond["operator"]
   405‚Üí   405‚Üí        val = cond.get("value", "")
   406‚Üí   406‚Üí        cell = str(row.get(col, ""))
   407‚Üí   407‚Üí
   408‚Üí   408‚Üí        if op == "equals":
   409‚Üí   409‚Üí            try:
   410‚Üí   410‚Üí                return float(cell) == float(val) if val != "" else cell == val
   411‚Üí   411‚Üí            except ValueError:
   412‚Üí   412‚Üí                return cell == val
   413‚Üí   413‚Üí        elif op == "not_equals":
   414‚Üí   414‚Üí            try:
   415‚Üí   415‚Üí                return float(cell) != float(val) if val != "" else cell != val
   416‚Üí   416‚Üí            except ValueError:
   417‚Üí   417‚Üí                return cell != val

   420‚Üí   420‚Üí        elif op == "not_contains":
   421‚Üí   421‚Üí            return val.lower() not in cell.lower()
   422‚Üí   422‚Üí        elif op == "starts_with":
   423‚Üí   423‚Üí            return cell.lower().startswith(val.lower())
   424‚Üí   424‚Üí        elif op == "ends_with":
   425‚Üí   425‚Üí            return cell.lower().endswith(val.lower())
   426‚Üí   426‚Üí        elif op == "greater_than":
   427‚Üí   427‚Üí            try:
   428‚Üí   428‚Üí                return float(cell) > float(val)
   429‚Üí   429‚Üí            except ValueError:
   430‚Üí   430‚Üí                return cell > val
   431‚Üí   431‚Üí        elif op == "less_than":
   432‚Üí   432‚Üí            try:
   433‚Üí   433‚Üí                return float(cell) < float(val)
   434‚Üí   434‚Üí            except ValueError:
   435‚Üí   435‚Üí                return cell < val
   436‚Üí   436‚Üí        elif op == "greater_or_equal":
   437‚Üí   437‚Üí            try:
   438‚Üí   438‚Üí                return float(cell) >= float(val)
   439‚Üí   439‚Üí            except ValueError:
   440‚Üí   440‚Üí                return cell >= val
   441‚Üí   441‚Üí        elif op == "less_or_equal":
   442‚Üí   442‚Üí            try:
   443‚Üí   443‚Üí                return float(cell) <= float(val)
   444‚Üí   444‚Üí            except ValueError:
   445‚Üí   445‚Üí                return cell <= val
   446‚Üí   446‚Üí        elif op == "is_empty":
   447‚Üí   447‚Üí            return cell.strip() == ""
   448‚Üí   448‚Üí        elif op == "is_not_empty":
   449‚Üí   449‚Üí            return cell.strip() != ""
   450‚Üí   450‚Üí        return True
   451‚Üí   451‚Üí
   452‚Üí   452‚Üí    dicts = df.to_dict("records")
   453‚Üí   453‚Üí    if combine_with == "OR":
   454‚Üí   454‚Üí        filtered = [r for r in dicts if any(apply_condition(r, c) for c in conditions)]
   455‚Üí   455‚Üí    else:
   456‚Üí   456‚Üí        filtered = [r for r in dicts if all(apply_condition(r, c) for c in conditions)]
   457‚Üí   457‚Üí
   458‚Üí   458‚Üí    filtered_df = pd.DataFrame(filtered) if filtered else pd.DataFrame(columns=df.columns)
   459‚Üí   459‚Üí    base_name = Path(filepath).stem
   460‚Üí   460‚Üí    info = save_df_to_file(filtered_df, base_name, "filtered")
   461‚Üí   461‚Üí    record_file(info["filename"], args.get("originalName", ""), info["mime"], info["size"], "filter", info["filepath"])
   462‚Üí   462‚Üí
   463‚Üí   463‚Üí    return {
   464‚Üí   464‚Üí        "success": True, "downloadUrl": f"/api/tools/download?file={info['filename']}",
   465‚Üí   465‚Üí        "filename": info["filename"], "totalRows": len(df),
   466‚Üí   466‚Üí        "matchedRows": len(filtered), "removedRows": len(df) - len(filtered),
   467‚Üí   467‚Üí        "conditions": conditions, "combineWith": combine_with, "preview": filtered[:10],
   468‚Üí   468‚Üí    }

   471‚Üí   471‚Üídef tool_replace(args):
   472‚Üí   472‚Üí    filepath = args.get("filepath", "")
   473‚Üí   473‚Üí    find = args.get("find", "")
   474‚Üí   474‚Üí    replace = args.get("replace", "")
   475‚Üí   475‚Üí    scope_columns = args.get("columns", [])
   476‚Üí   476‚Üí    match_mode = args.get("matchMode", "contains")
   477‚Üí   477‚Üí    case_sensitive = args.get("caseSensitive", "false") == "true"
   478‚Üí   478‚Üí    use_regex = args.get("useRegex", "false") == "true"
   479‚Üí   479‚Üí
   480‚Üí   480‚Üí    if not filepath:
   481‚Üí   481‚Üí        return {"error": "File is required"}
   482‚Üí   482‚Üí    if not find:
   483‚Üí   483‚Üí        return {"error": "Find text is required"}
   484‚Üí   484‚Üí
   485‚Üí   485‚Üí    df = read_file_to_df(filepath)
   486‚Üí   486‚Üí    if df.empty:
   487‚Üí   487‚Üí        return {"error": "File is empty"}
   488‚Üí   488‚Üí
   489‚Üí   489‚Üí    all_columns = list(df.columns)
   490‚Üí   490‚Üí    target_columns = [c for c in scope_columns if c in all_columns] if scope_columns else all_columns
   491‚Üí   491‚Üí    if not target_columns:
   492‚Üí   492‚Üí        return {"error": "No matching columns found"}
   493‚Üí   493‚Üí
   494‚Üí   494‚Üí    flags = 0 if case_sensitive else re.IGNORECASE
   495‚Üí   495‚Üí    try:
   496‚Üí   496‚Üí        if use_regex:
   497‚Üí   497‚Üí            pattern = re.compile(find, flags)
   498‚Üí   498‚Üí        else:
   499‚Üí   499‚Üí            escaped = re.escape(find)
   500‚Üí   500‚Üí            if match_mode == "exact":
   501‚Üí   501‚Üí                pattern = re.compile(f"^{escaped}$", flags)
   502‚Üí   502‚Üí            elif match_mode == "startsWith":
   503‚Üí   503‚Üí                pattern = re.compile(f"^{escaped}", flags)
   504‚Üí   504‚Üí            elif match_mode == "endsWith":
   505‚Üí   505‚Üí                pattern = re.compile(f"{escaped}$", flags)
   506‚Üí   506‚Üí            else:
   507‚Üí   507‚Üí                pattern = re.compile(escaped, flags)
   508‚Üí   508‚Üí    except re.error:
   509‚Üí   509‚Üí        return {"error": "Invalid regex pattern"}
   510‚Üí   510‚Üí
   511‚Üí   511‚Üí    total_matches = 0
   512‚Üí   512‚Üí    cells_changed = 0
   513‚Üí   513‚Üí    rows_affected = set()
   514‚Üí   514‚Üí    changes = []
   515‚Üí   515‚Üí
   516‚Üí   516‚Üí    for idx, row in df.iterrows():
   517‚Üí   517‚Üí        for col in target_columns:
   518‚Üí   518‚Üí            original = str(row[col])
   519‚Üí   519‚Üí            new_val, count = pattern.subn(replace, original)
   520‚Üí   520‚Üí            if count > 0:
   521‚Üí   521‚Üí                total_matches += count
   522‚Üí   522‚Üí                cells_changed += 1
   523‚Üí   523‚Üí                rows_affected.add(idx)
   524‚Üí   524‚Üí                df.at[idx, col] = new_val
   525‚Üí   525‚Üí                if len(changes) < 50:
   526‚Üí   526‚Üí                    changes.append({
   527‚Üí   527‚Üí                        "row": idx + 1, "column": col,
   528‚Üí   528‚Üí                        "before": original[:80], "after": new_val[:80],
   529‚Üí   529‚Üí                    })
   530‚Üí   530‚Üí
   531‚Üí   531‚Üí    base_name = Path(filepath).stem
   532‚Üí   532‚Üí    info = save_df_to_file(df, base_name, "replaced")
   533‚Üí   533‚Üí    record_file(info["filename"], args.get("originalName", ""), info["mime"], info["size"], "replace", info["filepath"])
   534‚Üí   534‚Üí
   535‚Üí   535‚Üí    return {
   536‚Üí   536‚Üí        "success": True, "downloadUrl": f"/api/tools/download?file={info['filename']}",
   537‚Üí   537‚Üí        "filename": info["filename"], "totalRows": len(df),
   538‚Üí   538‚Üí        "scopedColumns": target_columns, "matchMode": "regex" if use_regex else match_mode,
   539‚Üí   539‚Üí        "caseSensitive": case_sensitive, "totalMatches": total_matches,
   540‚Üí   540‚Üí        "cellsChanged": cells_changed, "rowsAffected": len(rows_affected),
   541‚Üí   541‚Üí        "changes": changes, "preview": df_to_dicts(df.head(10)),

   596‚Üí   596‚Üí    if not group_by:

   598‚Üí   598‚Üí    if not aggregations:

   612‚Üí   612‚Üí        col = a["column"]
   613‚Üí   613‚Üí        fn = a["function"]
   614‚Üí   614‚Üí        alias = a.get("alias") or f"{col}_{fn}"
   615‚Üí   615‚Üí        if fn != "count" and col not in df.columns:
   616‚Üí   616‚Üí            return {"error": f"Aggregation column '{col}' not found"}
   617‚Üí   617‚Üí
   618‚Üí   618‚Üí        pandas_fn = {"sum": "sum", "avg": "mean", "count": "count", "count_distinct": "nunique",
   619‚Üí   619‚Üí                     "min": "min", "max": "max", "first": "first", "last": "last"}.get(fn, "count")
   620‚Üí   620‚Üí
   621‚Üí   621‚Üí        key = f"{col}_{fn}_{alias}"
   622‚Üí   622‚Üí        agg_dict[key] = pd.NamedAgg(column=col, aggfunc=pandas_fn)
   623‚Üí   623‚Üí        rename_map[key] = alias
   624‚Üí   624‚Üí
   625‚Üí   625‚Üí    for a in aggregations:
   626‚Üí   626‚Üí        col = a["column"]
   627‚Üí   627‚Üí        if a["function"] in ("sum", "avg", "min", "max"):
   628‚Üí   628‚Üí            df[col] = pd.to_numeric(df[col], errors="coerce")
   629‚Üí   629‚Üí
   630‚Üí   630‚Üí    grouped = df.groupby(group_by, dropna=False).agg(**agg_dict).reset_index()
   631‚Üí   631‚Üí    grouped = grouped.rename(columns=rename_map)
   632‚Üí   632‚Üí
   633‚Üí   633‚Üí    first_col = group_by[0]
   634‚Üí   634‚Üí    try:
   635‚Üí   635‚Üí        grouped[first_col] = pd.to_numeric(grouped[first_col], errors="coerce")
   636‚Üí   636‚Üí    except Exception:
   637‚Üí   637‚Üí        pass
   638‚Üí   638‚Üí    grouped = grouped.sort_values(first_col, ascending=True, na_position="last").reset_index(drop=True)
   639‚Üí   639‚Üí
   640‚Üí   640‚Üí    base_name = Path(filepath).stem
   641‚Üí   641‚Üí    info = save_df_to_file(grouped, base_name, "pivot")
   642‚Üí   642‚Üí    record_file(info["filename"], args.get("originalName", ""), info["mime"], info["size"], "pivot", info["filepath"])
   643‚Üí   643‚Üí
   644‚Üí   644‚Üí    return {
   645‚Üí   645‚Üí        "success": True, "downloadUrl": f"/api/tools/download?file={info['filename']}",
   646‚Üí   646‚Üí        "filename": info["filename"], "totalRows": len(df),
   647‚Üí   647‚Üí        "groupCount": len(grouped), "groupBy": group_by,
   648‚Üí   648‚Üí        "aggregations": aggregations, "preview": df_to_dicts(grouped.head(20)),
   649‚Üí   649‚Üí    }

   730‚Üí   730‚Üí        if "date_format" in checks and col in date_cols:

   732‚Üí   732‚Üí                if v and not date_re.match(str(v)):

   740‚Üí   740‚Üí                continue
   741‚Üí   741‚Üí            if v not in seen:
   742‚Üí   742‚Üí                seen[v] = []
   743‚Üí   743‚Üí            seen[v].append(i + 1)

   745‚Üí   745‚Üí            if len(rows) > 1:

   760‚Üí   760‚Üí    }
   761‚Üí   761‚Üí    overall_score = max(0, round(100 - ((summary["errors"] * 5 + summary["warnings"] * 2 + summary["infos"] * 0.5) / max(1, total_cells)) * 100))
   762‚Üí   762‚Üí
   763‚Üí   763‚Üí    base_name = Path(filepath).stem

   766‚Üí   766‚Üí        summary_data = [{"Metric": k, "Value": v} for k, v in summary.items()] + [{"Metric": "Quality Score", "Value": f"{overall_score}/100"}]
